72 public:
75  inline bool operator==( const Vertex& v ) const {
76  switch ( type_ ) {
77  case Road:
78  return v.data_.vertex == data_.vertex;
79  case PublicTransport:
80  return v.data_.pt.index == data_.pt.index && v.data_.pt.vertex == data_.pt.vertex;
81  case Poi:
82  return v.data_.poi == data_.poi;
83  case Null:
84  return v.is_null();
85  }
86  return false;
87  }
88 
89  bool operator!=( const Vertex& v ) const { return !operator==( v ); }
90 
91  inline bool operator<( const Vertex& v ) const {
92  // Critical code, called in many places. Each minor optimisation is a huge gain
93  if (type() != v.type()) {
94  return type() < v.type();
95  }
96  switch (type_)
97  {
98  case Road:
99  return data_.vertex < v.data_.vertex;
100  case PublicTransport:
101  return data_.pt.index != v.data_.pt.index ? data_.pt.index < v.data_.pt.index : data_.pt.vertex < v.data_.pt.vertex;
102  case Poi:
103  return data_.poi < v.data_.poi;
104  case Null:
105  return false;
106  }
107  return false;
108  }
109 
110  inline int cmp( const Vertex& v ) const {
111  if (type() != v.type()) {
112  return type() < v.type() ? -1 : 1;
113  }
114  switch (type_)
115  {
116  case Road:
117  if ( data_.vertex != v.data_.vertex ) {
118  return data_.vertex < v.data_.vertex ? -1 : 1;
119  }
120  break;
121  case PublicTransport:
122  if ( data_.pt.index != v.data_.pt.index ) {
123  return data_.pt.index < v.data_.pt.index ? -1 : 1;
124  }
125  if ( data_.pt.vertex != v.data_.pt.vertex ) {
126  return data_.pt.vertex < v.data_.pt.vertex ? -1 : 1;
127  }
128  break;
129  case Poi:
130  if ( data_.poi != v.data_.poi ) {
131  return data_.poi < v.data_.poi ? -1 : 1;
132  }
133  default:
134  break;
135  }
136  return 0;
137  }
