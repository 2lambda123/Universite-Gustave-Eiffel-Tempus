140  std::map < Vertex, Vertex > build_failure_function() {
141  std::map< Vertex, Vertex > failure_function_;
142  failure_function_.clear();
143  std::list< Vertex > q;
144 
145  // All successors of the initial state return to initial state when failing
146  BGL_FORALL_OUTEDGES_T( initial_state_, edge, automaton_graph_, Graph ) {
147  q.push_back( target( edge, automaton_graph_ ) ) ;
148  failure_function_.insert( std::pair< Vertex, Vertex > ( target( edge, automaton_graph_ ), initial_state_ ) ) ;
149  }
150 
151  // Iterating over the list of states which do not have failure state
152  Vertex s, r ;
153 
154  while ( ! q.empty() ) {
155  s = q.front();
156  q.pop_front();
157  BGL_FORALL_OUTEDGES_T( s, edge, automaton_graph_, Graph ) {
158  q.push_back( target( edge, automaton_graph_ ) );
159  r = failure_function_[ s ];
160 
161  while ( find_transition( automaton_graph_, r, automaton_graph_[edge].entity ).second == false && r != initial_state_ ) {
162  r = failure_function_[ r ];
163  }
164 
165  if ( find_transition( automaton_graph_, r, automaton_graph_[edge].entity ).second == true ) {
166  failure_function_.insert( std::pair< Vertex, Vertex > ( s, find_transition( automaton_graph_, r, automaton_graph_[edge].entity ).first ) ) ;
167  }
168  else {
169  failure_function_.insert( std::pair< Vertex, Vertex > ( s, initial_state_ ) ) ;
170  }
171  }
172  }
