57 template <class RoadGraph>
58 double road_travel_time( const RoadGraph& road_graph, const Road::Edge& road_e, double length, double time, const TransportMode& mode,
59  double walking_speed = DEFAULT_WALKING_SPEED, double cycling_speed = DEFAULT_CYCLING_SPEED, const RoadEdgeSpeedProfile* profile = 0 )
60 {
61  if ( (road_graph[ road_e ].traffic_rules() & mode.traffic_rules()) == 0 ) { // Not allowed mode
62  return std::numeric_limits<double>::infinity() ;
63  }
64  if (profile) {
65  RoadEdgeSpeedProfile::PeriodIterator it, it_end;
66  bool found = false;
67  boost::tie(it, it_end) = profile->periods_after( road_graph[road_e].db_id(), mode.speed_rule(), time, found );
68  if ( found ) {
69  double t_end, t_begin, speed = 50000.0/60.0;
70  t_begin = time;
71  while ( (it != it_end) && (length > 0) ) {
72  speed = it->second.speed * 1000.0 / 60.0; // km/h -> m/min
73  t_end = it->first + it->second.length;
74  length -= speed * (t_end - t_begin);
75  t_begin = t_end;
76  it++;
77  }
78  return t_begin + (length / speed) - time;
79  }
80  }
81  return avg_road_travel_time( road_graph, road_e, length, mode, walking_speed, cycling_speed );
82 }

300  double transfer_time( const Multimodal::Edge& edge, const TransportMode& initial_mode, const TransportMode& final_mode ) const
301  {

311  return 0.0;
312  }
313 
314  // park shared vehicle
