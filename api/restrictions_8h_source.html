99  template <typename Progressor, typename PointCacheType>
100  void write_restrictions( const PointCacheType& points, Writer& writer, Progressor& progressor )
101  {
102  if ( restrictions_.size() == 0 )
103  return;
104 
105  progressor( 0, restrictions_.size() );
106  writer.begin_restrictions();
107  size_t ri = 0;
108  for ( const auto& tr : restrictions_ ) {
109  progressor( ++ri, restrictions_.size() );
110  // only process way - node - way relations
111  if ( points.find( tr.from_way ) ||
112  !points.find( tr.via_node ) ||
113  points.find( tr.to_way ) )
114  continue;
115  // get the first edge until on the "from" way
116  Section section_from;
117  Section section_to;
118  for ( auto section: way_sections_[tr.from_way] ) {
119  if ( section.node2 == tr.via_node ) {
120  section_from = section;
121  break;
122  }
123  if ( section.node1 == tr.via_node ) {
124  section_from.id = section.id;
125  section_from.node1 = section.node2;
126  section_from.node2 = section.node1;
127  break;
128  }
129  }
130  if ( section_from.id == 0 )
131  continue;
132 
133  std::vector<Section> sections_to;
134  for ( auto section: way_sections_[tr.to_way] ) {
135  if ( section.node1 == tr.via_node || section.node2 == tr.via_node )
136  sections_to.push_back( section );
137  }
138  if ( sections_to.size() == 1 ) {
139  section_to = sections_to[0];
140  //std::cout << "1 " << section_from.id << " via " << tr.via_node << " to " << section_to.id << std::endl;
141  writer.write_restriction( ++restriction_id, { section_from.id, section_to.id } );
142  }
143  else if ( sections_to.size() == 2 ) {
144  // choose left or right
145  float angle[2];
146  int i = 0;
147  for ( const auto& s : sections_to ) {
148  uint64_t n1 = s.node1;
149  uint64_t n2 = s.node2;
150  if ( tr.via_node == s.node2 )
151  std::swap( n1, n2 );
152  const auto& p1 = points.at( section_from.node1 );
153  const auto& p2 = points.at( tr.via_node );
154  const auto& p3 = points.at( n2 );
155  // compute the angle between the 3 points
156  // we could have used the orientation (determinant), but angle computation is more stable
157  angle[i] = angle_3_points( p1.lon(), p1.lat(), p2.lon(), p2.lat(), p3.lon(), p3.lat() );
158  i++;
159  }
160  if ( tr.type == TurnRestriction::NoLeftTurn || tr.type == TurnRestriction::OnlyLeftTurn ) {
161  // take the angle < 0
162  if ( angle[0] < 0 )
163  section_to = sections_to[0];
164  else
165  section_to = sections_to[1];
166  }
167  else if ( tr.type == TurnRestriction::NoRightTurn || tr.type == TurnRestriction::OnlyRightTurn ) {
168  // take the angle > 0
169  if ( angle[0] > 0 )
170  section_to = sections_to[0];
171  else
172  section_to = sections_to[1];
173  }
174  else if ( tr.type == TurnRestriction::NoStraightOn || tr.type == TurnRestriction::OnlyStraightOn ) {
175  // take the angle closer to 0
176  if ( std::abs(angle[0]) < std::abs(angle[1]) )
177  section_to = sections_to[0];
178  else
179  section_to = sections_to[1];
180  }
181  else {
182  std::cerr << "Ignoring restriction from " << tr.from_way << " to " << tr.to_way << " with type " << tr.type << std::endl;
183  continue;
184  }
185 
186  // now distinguish between NoXXX and OnlyXXX restriction types
187  if ( tr.type == TurnRestriction::NoLeftTurn ||
188  tr.type == TurnRestriction::NoRightTurn ||
189  tr.type == TurnRestriction::NoStraightOn ) {
190  // emit the restriction
191  //std::cout << "2 " << section_from.id << " via " << tr.via_node << " to " << section_to.id << std::endl;
192  writer.write_restriction( ++restriction_id, { section_from.id, section_to.id } );
193  }
194  else {
195  // OnlyXXX is like several NoXXX on the other edges
196  // we have to emit a restriction for every connected edges, except this one
197  for ( auto way : via_nodes_ways_.at( tr.via_node ) ) {
198  for ( auto section: way_sections_.at( way ) ) {
199  if ( section.node1 == tr.via_node || section.node2 == tr.via_node ) {
200  if ( section.id != section_to.id && section.id != section_from.id )
201  //std::cout << "3 " << section_from.id << " via " << tr.via_node << " to " << section.id << std::endl;
202  writer.write_restriction( ++restriction_id, { section_from.id, section.id } );
203  }
204  }
205  }
206  }
207 
208  }
209  }
210  writer.end_restrictions();
211  }
