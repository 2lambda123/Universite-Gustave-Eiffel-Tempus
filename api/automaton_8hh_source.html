162  std::map < State, State > build_failure_function()
163  {
164  std::cout << "failure function" << std::endl;
165  std::map< State, State > failure_function_;
166  failure_function_.clear();
167  std::list< State > q;
168 
169  // All successors of the initial state return to initial state when failing
170  BGL_FORALL_OUTEDGES_T( initial_state_, edge, automaton_graph_, Graph ) {
171  const State& s = target( edge, automaton_graph_ );
172  q.push_back( s );
173  failure_function_[ s ] = initial_state_;
174  }
175 
176  // Iterating over the list of states which do not have failure state
177  State r, state;
178  while ( ! q.empty() ) {
179  r = q.front();
180  q.pop_front();
181  BGL_FORALL_OUTEDGES_T( r, edge, automaton_graph_, Graph ) {
182  State s = target( edge, automaton_graph_ );
183  Symbol a = automaton_graph_[edge].symbol;
184  q.push_back( s );
185  state = failure_function_[ s ];
186  while ( find_transition( state, a ).second == false && state != initial_state_ ) {
187  state = failure_function_[ state ];
188  }
189  std::pair<State,bool> t = find_transition( state, a );
190  if ( t.second == true ) {
191  failure_function_[ s ] = t.first;
192  }
193  else {
194  failure_function_[ s ] = initial_state_;
195  }
196  }
197  }
198  std::cout << "end of failure" << std::endl;
199
