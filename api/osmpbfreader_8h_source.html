202  OSMPBF::BlobHeader read_header(){

204  OSMPBF::BlobHeader result;
205 
206  // read the first 4 bytes of the file, this is the size of the blob-header
207  if( !file.read((char*)&sz, 4) ){
208  this->finished = true;
209  return result;

227  int32_t read_blob(const OSMPBF::BlobHeader & header){
228  OSMPBF::Blob blob;
229  // size of the following blob
230  int32_t sz = header.datasize();
231 
232  if(sz > max_uncompressed_blob_size)
233  fatal() << "blob-size is bigger then allowed";
234 
235  if(!this->file.read(buffer, sz))
236  fatal() << "unable to read blob from file";
237  if(!blob.ParseFromArray(this->buffer, sz))
238  fatal() << "unable to parse blob";
239 
240  // if the blob has uncompressed data
241  if(blob.has_raw()) {
242  // size of the blob-data
243  sz = blob.raw().size();
244 
245  // check that raw_size is set correctly
246  if(sz != blob.raw_size())
247  warn() << " reports wrong raw_size: " << blob.raw_size() << " bytes";
248 
249  memcpy(unpack_buffer, buffer, sz);
250  return sz;
251  }
252 
253 
254  if(blob.has_zlib_data()) {
255  sz = blob.zlib_data().size();
256 
257  z_stream z;
258  z.next_in = (unsigned char*) blob.zlib_data().c_str();
259  z.avail_in = sz;
260  z.next_out = (unsigned char*) unpack_buffer;
261  z.avail_out = blob.raw_size();
262  z.zalloc = Z_NULL;
263  z.zfree = Z_NULL;
264  z.opaque = Z_NULL;
265 
266  if(inflateInit(&z) != Z_OK) {
267  fatal() << "failed to init zlib stream";
268  }
269  if(inflate(&z, Z_FINISH) != Z_STREAM_END) {
270  fatal() << "failed to inflate zlib stream";
271  }
272  if(inflateEnd(&z) != Z_OK) {
273  fatal() << "failed to deinit zlib stream";
274  }
275  return z.total_out;
276  }
277 
278  if(blob.has_lzma_data()) {
279  fatal() << "lzma-decompression is not supported";
280  }
281  return 0;
282  }

284  int enum_primitiveblock(int32_t sz) {
285  OSMPBF::PrimitiveBlock primblock;
286  if(!primblock.ParseFromArray(this->unpack_buffer, sz))
287  fatal() << "unable to parse primitive block";
288 
289  int block_types = 0;
290  for(int i = 0, l = primblock.primitivegroup_size(); i < l; i++) {
291  OSMPBF::PrimitiveGroup pg = primblock.primitivegroup(i);
292  if ( pg.relations_size() ) {
293  block_types |= RelationBlock;
294  }
295  if ( pg.ways_size() ) {
296  block_types |= WayBlock;
297  }
298  if ( pg.nodes_size() || pg.has_dense() ) {
299  block_types |= NodeBlock;
300  }
301  }
302  return block_types;
303  }

305  off_t next_block()
306  {
307  const char pattern[] = "\0\0\0\x0d\x0a\x07OSMData";
308  const size_t bsize = 13;
309  char buffer[bsize+1] = {0};
310  char c;
311  while (file.get(c)) {
312  memmove( &buffer[0], &buffer[1], bsize - 1 );
313  buffer[bsize-1] = c;
314  if ( memcmp( buffer, pattern, bsize ) == 0 ) {
315  file.seekg( file.tellg() - static_cast<off_t>( bsize ) );
316  return file.tellg();
317  }
318  }
319  return -1;
320  }

322  off_t binary_search_pattern_( off_t start, off_t end, BlockType block_type )
323  {
324  int iblock_type = static_cast<int>( block_type );
325  off_t half = (start + end) / 2;
326  if ( file.eof() )
327  file.clear();
328  file.seekg( half );
329  if ( next_block() == -1 )
330  return -1;
331 
332  int b1 = 0, b2 = 0;
333  OSMPBF::BlobHeader header = this->read_header();
334  if(!this->finished){
335  int32_t sz = this->read_blob(header);
336  if(header.type() == "OSMData") {
337  b1 = enum_primitiveblock( sz );
338  }
339  }
340  off_t found_off = file.tellg();
341  header = this->read_header();
342  if(!this->finished){
343  int32_t sz = this->read_blob(header);
344  if(header.type() == "OSMData") {
345  b2 = enum_primitiveblock( sz );
346  }
347  }
348  this->finished = false;
349  if ( ((b1 & iblock_type) == 0) && (b2 & iblock_type) ) {
350  return found_off;
351  }
352  else if ( b1 >= block_type ) {
353  return binary_search_pattern_( start, half, block_type );
354  }
355  else if ( b2 < block_type ) {
356  return binary_search_pattern_( half, end, block_type );
357  }
358  return -1;
359  }

436  void parse_primitiveblock(int32_t sz) {
437  OSMPBF::PrimitiveBlock primblock;
438  if(!primblock.ParseFromArray(this->unpack_buffer, sz))
439  fatal() << "unable to parse primitive block";
440 
441  for(int i = 0, l = primblock.primitivegroup_size(); i < l; i++) {
442  OSMPBF::PrimitiveGroup pg = primblock.primitivegroup(i);
443 
444  // Simple Nodes
445  for(int i = 0; i < pg.nodes_size(); ++i) {
446  OSMPBF::Node n = pg.nodes(i);
447  double lon = 0.000000001 * (primblock.lon_offset() + (primblock.granularity() * n.lon())) ;
448  double lat = 0.000000001 * (primblock.lat_offset() + (primblock.granularity() * n.lat())) ;
449  visitor.node_callback(n.id(), lon, lat, get_tags(n, primblock));
450  }
451 
452  // Dense Nodes
453  if(pg.has_dense()) {
454  OSMPBF::DenseNodes dn = pg.dense();
455  uint64_t id = 0;
456  double lon = 0;
457  double lat = 0;
458 
459  int current_kv = 0;
460 
461  for(int i = 0; i < dn.id_size(); ++i) {
462  id += dn.id(i);
463  lon += 0.000000001 * (primblock.lon_offset() + (primblock.granularity() * dn.lon(i)));
464  lat += 0.000000001 * (primblock.lat_offset() + (primblock.granularity() * dn.lat(i)));
465 
466  Tags tags;
467  while (current_kv < dn.keys_vals_size() && dn.keys_vals(current_kv) != 0){
468  uint64_t key = dn.keys_vals(current_kv);
469  uint64_t val = dn.keys_vals(current_kv + 1);
470  std::string key_string = primblock.stringtable().s(key);
471  std::string val_string = primblock.stringtable().s(val);
472  current_kv += 2;
473  tags[key_string] = val_string;
474  }
475  ++current_kv;
476  visitor.node_callback(id, lon, lat, tags);
477  }
478  }
479 
480  for(int i = 0; i < pg.ways_size(); ++i) {
481  OSMPBF::Way w = pg.ways(i);
482 
483  uint64_t ref = 0;
484  std::vector<uint64_t> refs;
485  for(int j = 0; j < w.refs_size(); ++j){
486  ref += w.refs(j);
487  refs.push_back(ref);
488  }
489  uint64_t id = w.id();
490  visitor.way_callback(id, get_tags(w, primblock), refs);
491  }
492 
493 
494  for(int i=0; i < pg.relations_size(); ++i){
495  OSMPBF::Relation rel = pg.relations(i);
496  uint64_t id = 0;
497  References refs;
498 
499  for(int l = 0; l < rel.memids_size(); ++l){
500  id += rel.memids(l);
501  refs.push_back(Reference(rel.types(l), id, primblock.stringtable().s(rel.roles_sid(l))));
502  }
503 
504  visitor.relation_callback(rel.id(), get_tags(rel, primblock), refs);
505  }
506  }
507  }
